# Использован язык программирования Python

'''Описание встречаемых в программе переменных:
f - файл, с которым ведётся работа
a - строка файла, с которым ведётся работа
product - модель устройства, задаваемая пользователем
res_list - список из кортежей, содержащих следующую информацию:
 название компании, тип устройства, разрешение экрана и цену
t - список, более удобный вариант представления строки файла
x - объект списка res_list, информация о котором в данный момент выводится
'''

while True:
    print('Введите модель устройства:', end=' ')
    product = input()
    if product == '666':
        break
    res_list = []
    with open('devices.csv') as f:
        for a in f.readlines()[1:]:  # [1:] чтобы исключить строчку с заголовком
            t = list(a.split(';'))
            if t[1] == product:
                res_list.append((t[0], t[2], t[3], t[8]))
    if len(res_list) == 0:
        print('У нас нет данного устройства.')
    else:
        print('По вашему запросу', product, 'найдены следующие варианты:')
        for x in res_list:
            print(x[0], product, '- тип устройства:', x[1] + '; Разрешение экрана -', x[2] + '; Цена -', x[3], end='')
